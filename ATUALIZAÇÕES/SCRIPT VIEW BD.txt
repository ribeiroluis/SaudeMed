BUSCAPORCEP

SELECT        dbo.ENDERECO.CEP, dbo.ENDERECO.ENDRECO, dbo.BAIRRO.BAIRRO, dbo.CIDADE.CIDADE, dbo.CIDADE.UF
FROM            dbo.ENDERECO INNER JOIN
                         dbo.BAIRRO ON dbo.ENDERECO.IDBAIRRO = dbo.BAIRRO.IDBAIRRO INNER JOIN
                         dbo.CIDADE ON dbo.ENDERECO.IDCIDADE = dbo.CIDADE.IDCIDADE AND dbo.BAIRRO.IDCIDADE = dbo.CIDADE.IDCIDADE


ViewTabelaFuncionarios

SELECT        dbo.FUNCIONARIO.IDFUNCIONARIO AS Id, dbo.FUNCIONARIO.NOME, dbo.FUNCIONARIO.CEP, dbo.ENDERECO.ENDRECO AS [End], 
                         dbo.FUNCIONARIO.NUMERO AS Nº, dbo.FUNCIONARIO.COMPLEMENTO AS Comp, dbo.BAIRRO.BAIRRO, dbo.CIDADE.CIDADE, 
                         dbo.FUNCIONARIO.TELEFONEFIXO AS [Tel Fixo], dbo.FUNCIONARIO.TELEFONECELULAR AS [Tel Celular], dbo.FUNCIONARIO.CPF, 
                         dbo.FUNCIONARIO.IDENTIDADE, dbo.FUNCIONARIO.DTNASCIMENTO AS [DT Nascimento], dbo.FUNCIONARIO.ADMISSAO AS [DT Admissao], 
                         dbo.FUNCIONARIO.FUNCAO AS Função, dbo.FUNCIONARIO.REGPROFISSIONAL AS [Reg Profissional]
FROM            dbo.FUNCIONARIO INNER JOIN
                         dbo.ENDERECO ON dbo.FUNCIONARIO.CEP = dbo.ENDERECO.CEP INNER JOIN
                         dbo.BAIRRO ON dbo.ENDERECO.IDBAIRRO = dbo.BAIRRO.IDBAIRRO INNER JOIN
                         dbo.CIDADE ON dbo.ENDERECO.IDCIDADE = dbo.CIDADE.IDCIDADE AND dbo.BAIRRO.IDCIDADE = dbo.CIDADE.IDCIDADE



ViewTabelaItensProduto

SELECT        dbo.ITEMPRODUTO.IDITEM, dbo.PRODUTO.CODBARRAS, dbo.PRODUTO.DESCRICAO, dbo.ITEMPRODUTO.LOTE, dbo.ITEMPRODUTO.VALIDADE, 
                         dbo.ITEMPRODUTO.QUANTIDADE, dbo.PRODUTO.PRECOCOMPRA, dbo.PRODUTO.PRECOVENDA
FROM            dbo.ITEMPRODUTO INNER JOIN
                         dbo.PRODUTO ON dbo.ITEMPRODUTO.IDPRODUTO = dbo.PRODUTO.IDPRODUTO
